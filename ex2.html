Create an Ensamble and then stack it.

models <- caretList(
  Survived ~ .,
  df_f,
  allowParallel=TRUE,
  trControl = 
  trainControl(
    method = "cv",
    number = 10,
    savePredictions = 'final',
    classProbs=TRUE,
    index=createFolds(df_f$Survived, 10)
  ),
  metric = metric,
  tuneList = list(
        C50 = caretModelSpec(
      method = "C5.0",
      tuneGrid = data.frame(
        .trials = 4,
        .model = 'tree',
        .winnow = TRUE
      )
    ),
    xgbTree = caretModelSpec(
      method = "xgbTree",
      tuneGrid = data.frame(
        .nrounds=20,
        .max_depth=c(4,6),
        .eta=0.3,
        .gamma=0.1,
        .colsample_bytree=0.5,
        .min_child_weight=0.01,
        .subsample=0.7
      )
    ),
    rf = caretModelSpec(
      method = "rf",
      tuneGrid = data.frame(.mtry=1)
    ),
    adaboost = caretModelSpec(
      method = "adaboost"
    )
  )
)
